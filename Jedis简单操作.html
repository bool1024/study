3.List（列表）类型实例 
  1. public static void main(String[] args) {
  2.     Jedis jedis = new Jedis("127.0.0.1", 6379);
  3.     jedis.auth("123456");
  4.     jedis.lpush("test-list", "Java");
  5.     jedis.lpush("test-list", "PHP");
  6.     jedis.lpush("test-list", "C++");
  7.     //-1表示最后一个
  8.     List<String> list = jedis.lrange("test-list", 0, -1);
  9.     for(String str: list){
  10.       System.out.println(str);
  11.     }
  12.     jedis.close();
  13. }

4.Hash（哈希）类型实例 
  1. public static void main(String[] args) {
  2.     Jedis jedis = new Jedis("127.0.0.1", 6379);
  3.     jedis.auth("123456");
  4.     Map<String, String> map = new HashMap<String, String>();
  5.     map.put("xiaoming", "man");
  6.     map.put("xiaohua", "man");
  7.     map.put("xiaohong", "woman");
  8.     jedis.hmset("test-hash", map);
  9.     Map<String, String> hashData = jedis.hgetAll("test-hash");
  10.     for(String key: hashData.keySet()){
  11.       System.out.println(hashData.get(key));
  12.     }
  13.     jedis.close();
  14. }

5.Set（无序，不重复集合）类型实例 
  1. public static void main(String[] args) {
  2.     Jedis jedis = new Jedis("127.0.0.1", 6379);
  3.     jedis.auth("123456");
  4.     jedis.sadd("test-set", "Java");
  5.     jedis.sadd("test-set", "PHP");
  6.     jedis.sadd("test-set", "C++");
  7.     Set<String> set = jedis.smembers("test-set");
  8.     for(String str: set){
  9.       System.out.println(str);
  10.     }
  11.     jedis.close();
  12. }

6.ZSet（有序，不重复集合）类型实例 
  1. public static void main(String[] args) {
  2.     Jedis jedis = new Jedis("127.0.0.1", 6379);
  3.     jedis.auth("123456");
  4.     jedis.zadd("test-zset", 1, "Java");
  5.     jedis.zadd("test-zset", 3, "PHP");
  6.     jedis.zadd("test-zset", 2, "C++");
  7.     jedis.zadd("test-zset", 2, "PHP");
  8.     Set<String> set = jedis.zrange("test-zset", 0, 3);
  9.     for(String str: set){
  10.       System.out.println(str);
  11.     }
  12.     jedis.close();
  13. }
注：zset每个存储元素都会关联一个double类型的数值，redis就是根据这个数值进行排序的，数值可以重复，但是存储元素不可以重复。
