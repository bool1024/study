  3. import java.io.StringReader;
  4. import java.io.StringWriter;
  6. import javax.xml.bind.JAXBContext;
  7. import javax.xml.bind.JAXBException;
  8. import javax.xml.bind.Marshaller;
  9. import javax.xml.bind.Unmarshaller;
  11. public class JaxbUtil {
  12.         
  13.         private JAXBContext jaxbContext;
  14.         
  15.         /**
  16.          * @param types 所有需要序列化的Root对象的类型
  17.          * 该方法为构造方法，进行JAXBContext的实例化使用，为后续编组，解组做准备
  18.          */
  19.         public JaxbUtil(Class<?>... types){
  20.                 try {
  21.                         jaxbContext = JAXBContext.newInstance(types);
  22.                 } catch (JAXBException e) {
  23.                         e.printStackTrace();
  24.                 }
  25.         }
  26.         
  27.         /**
  28.          * Java Object ->Xml
  29.          * 将对象转为Xml请求，该方法内部调用了marshal接口，使用了marshal接口的实例
  30.          * @param root
  31.          * @param encoding
  32.          * @return
  33.          */
  34.         public String toXml(Object root, String encoding){
  35.                 try {
  36.                         StringWriter writer = new StringWriter();
  37.                         createMarshaller(encoding).marshal(root, writer);
  38.                         return writer.toString().replace(" standalone=\"yes\"", "");
  39.                 } catch (JAXBException e) {
  40.                         e.printStackTrace();
  41.                 }
  42.                 return null;
  43.         }
  44.         
  45.         /**
  46.          * Xml -> Java Object
  47.          * 将Xml转换为Object，该方法内部调用了unmarshal接口，使用了unmarshal接口的实例
  48.          * @param xml
  49.          * @return
  50.          */
  51.         @SuppressWarnings("unchecked")
  52.         public <T> T fromXml(String xml){
  53.                 try {
  54.                         StringReader reader = new StringReader(xml);
  55.                         return (T)createUnmarshaller().unmarshal(reader);
  56.                 } catch (JAXBException e) {
  57.                         // TODO Auto-generated catch block
  58.                         e.printStackTrace();
  59.                 }
  60.                 return null;
  61.         }
  62.         
  63.         /**
  64.          * 创建Marshaller，设定encoding（可为null）
  65.          * @param encoding
  66.          * @return
  67.          */
  68.         private Marshaller createMarshaller(String encoding){
  69.                 try {
  70.                         Marshaller marshaller = jaxbContext.createMarshaller();
  71.                         marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
  72.                         if(null != encoding || !"".equals(encoding)){
  73.                                 marshaller.setProperty(Marshaller.JAXB_ENCODING, encoding);
  74.                         }
  75.                         return marshaller;
  76.                 } catch (JAXBException e) {
  77.                         e.printStackTrace();
  78.                 }
  79.                 return null;
  80.         }
  81.         
  82.         /**
  83.          * 创建UnMarshaller
  84.          */
  85.         private Unmarshaller createUnmarshaller(){
  86.                 try {
  87.                         return jaxbContext.createUnmarshaller();
  88.                 } catch (JAXBException e) {
  89.                         e.printStackTrace();
  90.                 }
  91.                 return null;
  92.         }
  93. }

备注：类中属性上边的注解必须设置正确，否则出现的结果将出乎意料或者报错，关键性的注解说明如下：
  • @XmlType：用在class类的注解，常与@XmlRootElement、@XmlAccessorType一起使用。它有三个属性：name、propOrder、namespace，经常使用的只有前两个属性。
    如上述所示： 
    @XmlType(name="RoomTypeVO", propOrder={"typeId", "typeName", "price"})
    用户指定该类的所有属性的排列顺序，如果不指定，则输出结果可能跟预期不符合，会按默认的字母排序。另外，propOrder属性必须列出全部属性，否则报错。
  • @XmlRootElement：用于类级别的注解，对应xml的根元素，常与@XmlType和@XmlAccessorType一起用。
    如上述所示： 
    @XmlRootElement(name="hotel")
    指定root根绝点名称
  • @XmlAccessorType：用于指定由java对象生成xml文件时对java对象属性的访问方式。常与@XmlRootElement、@XmlType一起使用。实例如下： 
    @XmlAccessorType(XmlAccessType.FIELD);
    它的属性值是XmlAccessType的4个枚举值，分别为：
    • XmlAccessType.FIELD：java对象中的所有成员变量
    • XmlAccessType.PROPERTY：java对象中所有通过getter/setter方式访问的成员变量
    • XmlAccessType.PUBLIC_MEMBER：java对象中所有的public访问权限的成员变量和通过getter/setter方式访问的成员变量
    • XmlAccessType.NONE：java对象的所有属性都不映射为xml元素
    默认访问级别是XmlAccessType.PUBLIC_MEMBER，因此，如果java对象中的private成员变量设置了public权限的getter/setter方法，就不要在private变量上使用@XmlElement和@XmlAttribute注解，否则在由java对象生成xml时会报同一个属性在java类里存在两次的错误。同理，如果@XmlAccessorType的访问权限为XmlAccessType.NONE，如果在java的成员变量上使用了@XmlElement或@XmlAttribute注解，这些成员变量依然可以映射到xml文件。
  • @XmlElement：将java对象的属性映射为xml的节点，在使用@XmlElement时，可通过name属性改变java对象属性在xml中显示的名称。如上所示： 
    @XmlElement(name="price")
public String getPrice(){
    return price;
}
    如果不指定，则会输出符合java的属性，比如上述的如果指定name="price1"，则xml会输出price1
  • @XmlAttribute：用于把java对象的属性映射为xml的属性，并可通过name属性为生成的xml属性指定别名，如上所示： 
    @XmlAttribute(name="id")
public int getId(){
    return id;
}
    如果指定的属性为id1，则对应的attribute属性为id1
