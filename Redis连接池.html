需要jeids-2.8.jar包以及commons-pool2-2.4.2.jar包

1.src文件夹下新建redis.properties文件，内容如下 
#Redis settings
redis.host=127.0.0.1
redis.port=6379
redis.pass=123456
redis.timeout=10000

#max idle connection
redis.maxIdle=300
#max connection
redis.maxTotal=600

redis.maxWaitMillis=1000
redis.testOnBorrow=false

2.新建PropertyUtil.java文件，代码如下： 
  1. package com.java.redis.utils;
  3. import java.io.IOException;
  4. import java.io.InputStream;
  5. import java.util.Properties;
  7. public class PropertyUtil {
  8.   
  9.   /**
  10.    * 加载property文件到io流
  11.    */
  12.   public static Properties loadProperties(String propertyFile){
  13.     Properties properties = new Properties();
  14.     try {
  15.       InputStream inputStream = PropertyUtil.class.getClassLoader().getResourceAsStream(propertyFile);
  16.       if(inputStream == null){
  17.         inputStream = PropertyUtil.class.getClassLoader().getResourceAsStream("properties/" + propertyFile);
  18.       }
  19.       properties.load(inputStream);
  20.     } catch (IOException e) {
  21.       e.printStackTrace();
  22.     }
  23.     return properties;
  24.   }
  25.   
  26.   /**
  27.    * 根据key值取得对应value的值
  28.    */
  29.   public static String getValue(String propertyFile, String key){
  30.     Properties properties = loadProperties(propertyFile);
  31.     return properties.getProperty(key);
  32.   }
  33. }

3.新建JedisUtil.java文件，代码如下： 
  1. package com.java.redis.utils;
  3. import java.util.Properties;
  5. import redis.clients.jedis.Jedis;
  6. import redis.clients.jedis.JedisPool;
  7. import redis.clients.jedis.JedisPoolConfig;
  9. public class JedisUtil {
  10.   private static JedisPool jedisPool = null;
  12.   private static JedisUtil instance = null;
  14.   private JedisUtil() {
  16.   }
  18.   static {
  19.     Properties properties = PropertyUtil.loadProperties("redis.properties");
  20.     String host = properties.getProperty("redis.host");
  21.     String port = properties.getProperty("redis.port");
  22.     String pass = properties.getProperty("redis.pass");
  23.     String timeout = properties.getProperty("redis.timeout");
  24.     String maxIdle = properties.getProperty("redis.maxIdle");
  25.     String maxTotal = properties.getProperty("redis.maxTotal");
  26.     String maxWaitMillis = properties.getProperty("redis.maxWaitMillis");
  27.     String testOnBorrow = properties.getProperty("redis.testOnBorrow");
  28.     System.out.println(testOnBorrow);
  30.     JedisPoolConfig config = new JedisPoolConfig();
  31.     // 控制一个pool可分配多个jedis实例,通过pool.getResource()来获取
  32.     // 如果赋值为-1，则表示不限制；如果pool已经分配了maxActive个jedis实例，则此时pool的状态为exhausted（耗尽）
  33.     config.setMaxTotal(Integer.parseInt(maxTotal));
  34.     // 控制一个pool最多有多个状态为idle（空闲的）的jedis实例
  35.     config.setMaxIdle(Integer.parseInt(maxIdle));
  36.     // 表示当borrow（引入）一个jedis实例时，最大的等待时间，如果操作等待时间，则直接抛出JedisConnectionException
  37.     config.setMaxWaitMillis(Long.parseLong(maxWaitMillis));
  38.     // 在borrow一个jedis实例时，是否提前进行validate操作；如果为true，则得到的jedis实例均是可用的
  39.     config.setTestOnBorrow(Boolean.valueOf(testOnBorrow));
  41.     jedisPool = new JedisPool(config, host, Integer.parseInt(port), Integer.parseInt(timeout), pass);
  42.   }
  44.   /**
  45.    * 获取JedisUtil实例
  46.    */
  47.   public static JedisUtil getInstance() {
  48.     if (instance == null) {
  49.       synchronized (JedisUtil.class) {
  50.         if (instance == null) {
  51.           instance = new JedisUtil();
  52.         }
  53.       }
  54.     }
  55.     return instance;
  56.   }
  58.   /**
  59.    * 从jedis连接池中获取jedis对象
  60.    */
  61.   public Jedis getJedis() {
  62.     return jedisPool.getResource();
  63.   }
  64. }

4.使用方法 
  1. public static void main(String[] args) {
  2.     Jedis jedis = JedisUtil.getInstance().getJedis();
  3.     jedis.close();
  4. }


注：commons-pool2.jar文件中源码GenericObjectPool.java类内部的setConfig方法说明：
  • maxActive：指明能从池中借出的对象的最大数目（能够同时建立的最大连接个数）。如果这个值不是整数，表示没有限制。
  • maxIdle：最大空闲数（空闲连接数大于maxIdle时将进行回收）
  • minIdle：最小空闲数（低于minIdle时将创建新的连接）
  • whenExhaustedAction：指定在池中借出对象的数目已达极限的情况下，调用它的borrowObject()方法时的行为。可选值有：
    • GenericObjectPool.WHEN_EXHAUSTED_BLOCK，表示等待
    • GenericObjectPool.WHEN_EXHAUSTED_GROW，表示创建新的实例（不过这就使maxActive参数失去了意义）
    • GenericObjectPool.WHEN_EXHAUSTED_FAIL，表示抛出一个java.util.NoSuchElementException异常
  • maxWait：指明若在对象池空间时调用borrowObject()方法的行为被设定成等待，最多等待多少毫秒。如果等待时间超过了这个数值，则会抛出java.util.NoSuchElementException异常，如果这个值不是正数，表示无期限等待。
  • testOnBorrow：设定在借出对象时是否进行有效性检查。
  • testOnReturn：设定在还回对象时是否进行有效性检查。
  • timeBetweenEvictionRunsMillis：设定间隔每过多少毫秒进行一次后台对象清理的行动。如果这个值不是正数，则实际上不会进行后台对象清理。
  • minEvictableIdleTimeMillis：设定在进行后台对象清理时，视休眠时间超过了多少毫秒的对象为过期。过期的对象将被回收。如果这个值不是正数，那么对休眠时间没有特别的约束。
  • testWhileIdle：设定在进行后台对象清理时，是否还对没有过期的池内对象进行有效性检查。不能通过有效性检查的对象也将被回收。
  • lifo：池对象的放入和去除默认是后进先出的原则，默认是true，代表后进先出，设置为false代表先进先出。
