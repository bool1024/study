6.配置mapper.xml配置文件 
  1. <?xml version="1.0" encoding="UTF-8" ?>
  2. <!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
  3. "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
  4. <!-- 映射器 -->
  5. <!-- 命名空间：绑定接口，最好是使用完全限定名 -->
  6. <mapper namespace="com.java.mybatis.dao.IStudentDao">
  7.   <!-- 查询 -->
  8.   <!-- id：对应接口的方法名；parameterType：参数类型；resultMap：自定义结果集 -->
  9.   <!-- 该查询时根据学生id，查询该学生的基本信息以及其选课的课程信息和对应的老师信息，采用的是左外连接的方式 -->
  10.   <select id="getStudent" parameterType="integer" resultMap="StudentMap">
  11.     select s.id as studentid,s.name as studentname,c.name as coursename, age, c.id as courseid, t.id as teacherid, t.name as teachername
  12.     from student s left outer join StudentCourse sc on s.id=sc.student_id left outer join course c on c.id=sc.course_id left outer join 
  13.     teacher t on t.id=c.teacher_id where s.id=#{id};
  14.   </select>
  15.   
  16.   <!-- 自定义结果集 -->
  17.   <!-- type：结果集类型（type采用的是昵称，而不是完全限定名） -->
  18.   <resultMap id="StudentMap" type="Student">
  19.     <!-- 通过构造器来构造Student对象 -->
  20.     <!-- 带有该参数列表的构造器，必须在Bean中存在 -->
  21.     <!-- MyBatis框架会按名自动注入 -->
  22.     <constructor>
  23.       <!-- idArg：标记结果作为id可以帮助提高整体效能，一般是标记主键 -->
  24.                 <!-- 调用的是public Student(Integer id, String name, Integer age)构造函数，不是public Student(String name, Integer age)构造函数 -->
  25.       <idArg column="studentId" javaType="Integer" />
  26.       <!-- arg：注入到构造方法的一个普通结果 -->
  27.       <arg column="studentName" javaType="String" />
  28.       <arg column="age" javaType="Integer" />
  29.     </constructor>
  30.     
  31.     <!-- 集合，实现关系模型到对象模型的一对多映射 -->
  32.     <!-- property：必须为其上层元素代表的对象的一个集合属性，并且名字必须一样 -->
  33.     <!-- ofType：集合中元素的类型 -->
  34.     <!-- PS：如果该元素添加javaType属性，其值将是集合类型，比如说ArrayList -->
  35.     <collection property="courses" ofType="Course">
  36.       <!-- id：标记结果作为id可以帮助提高整体效能，一般是标记主键 -->
  37.       <id property="id" column="courseId" />
  38.       <!-- result：注入的一个普通结果 -->
  39.       <result property="name" column="courseName" />
  40.       
  41.       <!-- 关联，实现关系模型到对象模型的一对一的映射 -->
  42.       <!-- property：必须为其上层元素代表的对象的一个属性，并且名字必须一样 -->
  43.       <!-- javaType：属性类型 -->
  44.       <association property="teacher" javaType="Teacher">
  45.         <constructor>
  46.           <idArg column="teacherId" javaType="Integer" />
  47.           <arg column="teacherName" javaType="String" />
  48.         </constructor>
  49.       </association>
  50.     </collection>
  51.   </resultMap>
  52. </mapper>
